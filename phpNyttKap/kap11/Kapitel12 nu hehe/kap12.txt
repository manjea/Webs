uppg2
{
    SELECT country.Name, city.Name, city.Population, countrylanguage.Language 
    FROM country 

    INNER JOIN city 
    ON city.CountryCode = country.Code 

    INNER JOIN countrylanguage 
    ON countrylanguage.CountryCode = country.Code;
}
